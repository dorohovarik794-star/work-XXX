 Генерація секретних номерів
vectorint generateSecret() {
    vectorint secret(4);
    for (int i = 0; i  4; i++) {
        int d;
        bool ok;
        do {
            d = rand() % 10;
            ok = true;
            for (int j = 0; j  i; j++)
                if (secret[j] == d)
                    ok = false;
        } while (!ok);
        secret[i] = d;
    }
    return secret;
}

 Підрахунок биків та корів
pairint, int getBullsCows(const vectorint& secret, const string& guess) {
    int bulls = 0, cows = 0;
    for (int i = 0; i  4; i++) {
        int g = guess[i] - '0';
        if (secret[i] == g)
            bulls++;
        else {
            for (int j = 0; j  4; j++)
                if (secret[j] == g)
                    cows++;
        }
    }
    return { bulls, cows };
}
